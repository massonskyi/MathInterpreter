899 void reduce()
900 {
901 // Проверка деления на ноль
902 if (den_.getValue<int>() == 0)
903 {
904 throw std::runtime_error("Division by zero");
905 }
906
907 // Если числитель и знаменатель являются целыми числами
908 if constexpr (std::is_integral_v<decltype(num_.getValue<int>())>)
909 {
910 // Нормализация отрицательного знаменателя
911 if (den_.getValue<int>() < 0)
912 {
913 num_.set(-num_.getValue<int>());
914 den_.set(-den_.getValue<int>());
915 }
916
917 // Сокращение дроби с использованием GCD
918 if (int gcd = std::gcd(std::abs(num_.getValue<int>()), std::abs(den_.getValue<int>())); gcd > 1)
919 {
920 num_.set(num_.getValue<int>() / gcd);
921 den_.set(den_.getValue<int>() / gcd);
922 }
923 }
924 else if constexpr (std::is_floating_point_v<decltype(num_.getValue<double>())>)
925 {
926 // Нормализация для вещественных чисел
927 if (std::abs(den_.getValue<double>()) < std::numeric_limits<double>::epsilon())
928 {
929 throw std::runtime_error("Division by zero");
930 }
931
932 auto normalize = den_.getValue<double>();
933 num_.set(num_.getValue<double>() / normalize);
934 den_.set(1.0);
935 }
